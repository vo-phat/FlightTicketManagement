SET time_zone = '+07:00';
SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS `Baggage_History`;
DROP TABLE IF EXISTS `Baggage`;
DROP TABLE IF EXISTS `Payments`;
DROP TABLE IF EXISTS `Ticket_History`;
DROP TABLE IF EXISTS `Tickets`;
DROP TABLE IF EXISTS `Booking_Passengers`;
DROP TABLE IF EXISTS `Passenger_Profiles`;
DROP TABLE IF EXISTS `Bookings`;
DROP TABLE IF EXISTS `Fare_Rules`;
DROP TABLE IF EXISTS `Flight_Seats`;
DROP TABLE IF EXISTS `Seats`;
DROP TABLE IF EXISTS `Cabin_Classes`;
DROP TABLE IF EXISTS `Flights`;
DROP TABLE IF EXISTS `Routes`;
DROP TABLE IF EXISTS `Aircrafts`;
DROP TABLE IF EXISTS `Airports`;
DROP TABLE IF EXISTS `Airlines`;
DROP TABLE IF EXISTS `User_Role`;
DROP TABLE IF EXISTS `Roles`;
DROP TABLE IF EXISTS `Accounts`;
SET FOREIGN_KEY_CHECKS = 1;

-- ========================= 1. Accounts ===============================
CREATE TABLE `Accounts` (
    `account_id` INT AUTO_INCREMENT PRIMARY KEY,
    `email`      VARCHAR(100) NOT NULL UNIQUE,
    `password`   VARCHAR(255) NOT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========================= 2. Roles =================================
CREATE TABLE `Roles` (
    `role_id`   INT AUTO_INCREMENT PRIMARY KEY,
    `role_name` VARCHAR(50) NOT NULL UNIQUE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========================= 3. User_Role (N-N) =======================
CREATE TABLE `User_Role` (
    `user_role_id` INT AUTO_INCREMENT PRIMARY KEY,
    `account_id`   INT NOT NULL,
    `role_id`      INT NOT NULL,
    CONSTRAINT `fk_userrole_account`
        FOREIGN KEY (`account_id`) REFERENCES `Accounts`(`account_id`)
        ON UPDATE CASCADE ON DELETE CASCADE,
    CONSTRAINT `fk_userrole_role`
        FOREIGN KEY (`role_id`) REFERENCES `Roles`(`role_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_userrole_account` ON `User_Role` (`account_id`);
CREATE INDEX `idx_userrole_role`    ON `User_Role` (`role_id`);

-- ========================= 4. Airlines ==============================
CREATE TABLE `Airlines` (
    `airline_id`   INT AUTO_INCREMENT PRIMARY KEY,
    `airline_code` VARCHAR(10) NOT NULL UNIQUE,
    `airline_name` VARCHAR(100) NOT NULL,
    `country`      VARCHAR(100)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========================= 5. Airports ==============================
CREATE TABLE `Airports` (
    `airport_id`   INT AUTO_INCREMENT PRIMARY KEY,
    `airport_code` VARCHAR(10) NOT NULL UNIQUE,
    `airport_name` VARCHAR(100) NOT NULL,
    `city`         VARCHAR(100),
    `country`      VARCHAR(100)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ========================= 6. Aircrafts =============================
CREATE TABLE `Aircrafts` (
    `aircraft_id`  INT AUTO_INCREMENT PRIMARY KEY,
    `airline_id`   INT NOT NULL,
    `model`        VARCHAR(100),
    `manufacturer` VARCHAR(100),
    `capacity`     INT,
    CONSTRAINT `fk_aircraft_airline`
        FOREIGN KEY (`airline_id`) REFERENCES `Airlines`(`airline_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_aircraft_airline` ON `Aircrafts` (`airline_id`);

-- ========================= 7. Routes ================================
CREATE TABLE `Routes` (
    `route_id`          INT AUTO_INCREMENT PRIMARY KEY,
    `departure_place_id` INT NOT NULL,
    `arrival_place_id`   INT NOT NULL,
    `distance_km`       INT,
    `duration_minutes`  INT,
    CONSTRAINT `fk_routes_dep`
        FOREIGN KEY (`departure_place_id`) REFERENCES `Airports`(`airport_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT `fk_routes_arr`
        FOREIGN KEY (`arrival_place_id`) REFERENCES `Airports`(`airport_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT `chk_routes_diff` CHECK (`departure_place_id` <> `arrival_place_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_routes_dep` ON `Routes` (`departure_place_id`);
CREATE INDEX `idx_routes_arr` ON `Routes` (`arrival_place_id`);

-- ========================= 8. Flights ===============================
CREATE TABLE `Flights` (
    `flight_id`     INT AUTO_INCREMENT PRIMARY KEY,
    `flight_number` VARCHAR(20) NOT NULL,
    `aircraft_id`   INT NOT NULL,
    `route_id`      INT NOT NULL,
    `departure_time` DATETIME,
    `arrival_time`   DATETIME,
    `status` ENUM('SCHEDULED','DELAYED','CANCELLED','COMPLETED') NOT NULL DEFAULT 'SCHEDULED',
    CONSTRAINT `fk_flight_aircraft`
        FOREIGN KEY (`aircraft_id`) REFERENCES `Aircrafts`(`aircraft_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT `fk_flight_route`
        FOREIGN KEY (`route_id`) REFERENCES `Routes`(`route_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX `ux_flight_number_time` ON `Flights` (`flight_number`, `departure_time`);
CREATE INDEX `idx_flight_aircraft` ON `Flights` (`aircraft_id`);
CREATE INDEX `idx_flight_route`    ON `Flights` (`route_id`);

-- ========================= 9. Cabin_Classes =========================
CREATE TABLE `Cabin_Classes` (
    `class_id`    INT AUTO_INCREMENT PRIMARY KEY,
    `class_name`  VARCHAR(50) NOT NULL, -- Economy, Business, First
    `description` VARCHAR(255)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX `ux_cabin_class_name` ON `Cabin_Classes` (`class_name`);

-- ========================= 10. Seats (định danh ghế) ================
CREATE TABLE `Seats` (
    `seat_id`     INT AUTO_INCREMENT PRIMARY KEY,
    `aircraft_id` INT NOT NULL,
    `seat_number` VARCHAR(10) NOT NULL,
    `class_id`    INT NOT NULL,
    CONSTRAINT `fk_seat_aircraft`
        FOREIGN KEY (`aircraft_id`) REFERENCES `Aircrafts`(`aircraft_id`)
        ON UPDATE CASCADE ON DELETE CASCADE,  -- xóa máy bay sẽ xóa ghế
    CONSTRAINT `fk_seat_class`
        FOREIGN KEY (`class_id`) REFERENCES `Cabin_Classes`(`class_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX `ux_seat_aircraft_number` ON `Seats` (`aircraft_id`, `seat_number`);
CREATE INDEX `idx_seat_class` ON `Seats` (`class_id`);

-- ========================= 11. Flight_Seats (ghế theo chuyến) =======
CREATE TABLE `Flight_Seats` (
    `flight_seat_id` INT AUTO_INCREMENT PRIMARY KEY,
    `flight_id`      INT NOT NULL,
    `seat_id`        INT NOT NULL,
    `base_price`     DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    `seat_status`    ENUM('AVAILABLE','BOOKED','BLOCKED') NOT NULL DEFAULT 'AVAILABLE',
    CONSTRAINT `fk_fseat_flight`
        FOREIGN KEY (`flight_id`) REFERENCES `Flights`(`flight_id`)
        ON UPDATE CASCADE ON DELETE CASCADE,
    CONSTRAINT `fk_fseat_seat`
        FOREIGN KEY (`seat_id`) REFERENCES `Seats`(`seat_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX `ux_fseat_flight_seat` ON `Flight_Seats` (`flight_id`, `seat_id`);
CREATE INDEX `idx_fseat_status` ON `Flight_Seats` (`seat_status`);

-- ========================= 12. Fare_Rules ===========================
CREATE TABLE `Fare_Rules` (
    `rule_id`        INT AUTO_INCREMENT PRIMARY KEY,
    `route_id`       INT NOT NULL,
    `class_id`       INT NOT NULL,
    `fare_type`      VARCHAR(50) NOT NULL DEFAULT 'Standard', -- Saver, Flex, Promo...
    `season`         ENUM('PEAK','OFFPEAK','NORMAL') NOT NULL DEFAULT 'NORMAL',
    `effective_date` DATE,
    `expiry_date`    DATE,
    `description`    VARCHAR(255),
    `price`          DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    CONSTRAINT `fk_fare_route`
        FOREIGN KEY (`route_id`) REFERENCES `Routes`(`route_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT `fk_fare_class`
        FOREIGN KEY (`class_id`) REFERENCES `Cabin_Classes`(`class_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_fare_route_class` ON `Fare_Rules` (`route_id`, `class_id`);

-- ========================= 13. Bookings =============================
CREATE TABLE `Bookings` (
    `booking_id`   INT AUTO_INCREMENT PRIMARY KEY,
    `account_id`   INT NOT NULL,
    `booking_date` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `status`       ENUM('PENDING','CONFIRMED','CANCELLED','REFUNDED') NOT NULL DEFAULT 'PENDING',
    `total_amount` DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    CONSTRAINT `fk_booking_account`
        FOREIGN KEY (`account_id`) REFERENCES `Accounts`(`account_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_booking_account` ON `Bookings` (`account_id`);
CREATE INDEX `idx_booking_status`  ON `Bookings` (`status`);

-- ========================= 14. Passenger_Profiles ===================
CREATE TABLE `Passenger_Profiles` (
    `profile_id`     INT AUTO_INCREMENT PRIMARY KEY,
    `account_id`     INT NOT NULL,
    `full_name`      VARCHAR(100) NOT NULL,
    `date_of_birth`  DATE,
    `phone_number`   VARCHAR(255),
    `passport_number` VARCHAR(50),
    `nationality`     VARCHAR(50),
    CONSTRAINT `fk_profile_account`
        FOREIGN KEY (`account_id`) REFERENCES `Accounts`(`account_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_profile_account` ON `Passenger_Profiles` (`account_id`);
CREATE INDEX `idx_profile_passport` ON `Passenger_Profiles` (`passport_number`);

-- ========================= 15. Booking_Passengers ===================
CREATE TABLE `Booking_Passengers` (
    `booking_passenger_id` INT AUTO_INCREMENT PRIMARY KEY,
    `booking_id`           INT NOT NULL,
    `profile_id`           INT NOT NULL,
    CONSTRAINT `fk_bp_booking`
        FOREIGN KEY (`booking_id`) REFERENCES `Bookings`(`booking_id`)
        ON UPDATE CASCADE ON DELETE CASCADE,
    CONSTRAINT `fk_bp_profile`
        FOREIGN KEY (`profile_id`) REFERENCES `Passenger_Profiles`(`profile_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX `ux_bp_booking_profile` ON `Booking_Passengers` (`booking_id`, `profile_id`);

-- ========================= 16. Tickets ==============================
CREATE TABLE `Tickets` (
    `ticket_id`          INT AUTO_INCREMENT PRIMARY KEY,
    `ticket_passenger_id` INT NOT NULL,
    `flight_seat_id`     INT NOT NULL,
    `ticket_number`      VARCHAR(50) UNIQUE,
    `issue_date`         DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `status` ENUM('BOOKED','CONFIRMED','CHECKED_IN','BOARDED','CANCELLED','REFUNDED') NOT NULL DEFAULT 'BOOKED',
    CONSTRAINT `fk_ticket_bp`
        FOREIGN KEY (`ticket_passenger_id`) REFERENCES `Booking_Passengers`(`booking_passenger_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT `fk_ticket_fseat`
        FOREIGN KEY (`flight_seat_id`) REFERENCES `Flight_Seats`(`flight_seat_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE UNIQUE INDEX `ux_ticket_fseat` ON `Tickets` (`flight_seat_id`);
CREATE INDEX `idx_ticket_status`    ON `Tickets` (`status`);

-- ========================= 17. Ticket_History =======================
CREATE TABLE `Ticket_History` (
    `history_id` INT AUTO_INCREMENT PRIMARY KEY,
    `ticket_id`  INT NOT NULL,
    `old_status` ENUM('BOOKED','CONFIRMED','CHECKED_IN','BOARDED','CANCELLED','REFUNDED'),
    `new_status` ENUM('BOOKED','CONFIRMED','CHECKED_IN','BOARDED','CANCELLED','REFUNDED'),
    `changed_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT `fk_th_ticket`
        FOREIGN KEY (`ticket_id`) REFERENCES `Tickets`(`ticket_id`)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_th_ticket` ON `Ticket_History` (`ticket_id`);

-- ========================= 18. Payments =============================
CREATE TABLE `Payments` (
    `payment_id`     INT AUTO_INCREMENT PRIMARY KEY,
    `booking_id`     INT NOT NULL,
    `amount`         DECIMAL(12,2) NOT NULL,
    `payment_method` ENUM('CREDIT_CARD','BANK_TRANSFER','E_WALLET','CASH') NOT NULL,
    `payment_date`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `status`         ENUM('SUCCESS','FAILED','PENDING') NOT NULL DEFAULT 'PENDING',
    CONSTRAINT `fk_payment_booking`
        FOREIGN KEY (`booking_id`) REFERENCES `Bookings`(`booking_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_payment_booking` ON `Payments` (`booking_id`);
CREATE INDEX `idx_payment_status`  ON `Payments` (`status`);

-- ========================= 19. Baggage ==============================
CREATE TABLE `Baggage` (
    `baggage_id`         INT AUTO_INCREMENT PRIMARY KEY,
    `ticket_id`          INT NOT NULL,
    `flight_id`          INT NOT NULL,
    `baggage_tag`        VARCHAR(20) UNIQUE,
    `baggage_type`       ENUM('CHECKED','CARRY_ON','SPECIAL') NOT NULL DEFAULT 'CHECKED',
    `weight_kg`          DECIMAL(5,2) NOT NULL,
    `allowed_weight_kg`  DECIMAL(5,2) NOT NULL,
    `fee`                DECIMAL(12,2) NOT NULL DEFAULT 0.00,
    `status`             ENUM('CREATED','CHECKED_IN','LOADED','IN_TRANSIT','CLAIMED','LOST') NOT NULL DEFAULT 'CREATED',
    `special_handling`   VARCHAR(100),
    `created_at`         DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at`         DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT `fk_baggage_ticket`
        FOREIGN KEY (`ticket_id`) REFERENCES `Tickets`(`ticket_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT,
    CONSTRAINT `fk_baggage_flight`
        FOREIGN KEY (`flight_id`) REFERENCES `Flights`(`flight_id`)
        ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_baggage_ticket` ON `Baggage` (`ticket_id`);
CREATE INDEX `idx_baggage_flight` ON `Baggage` (`flight_id`);
CREATE INDEX `idx_baggage_status` ON `Baggage` (`status`);

-- ========================= 20. Baggage_History ======================
CREATE TABLE `Baggage_History` (
    `history_id` INT AUTO_INCREMENT PRIMARY KEY,
    `baggage_id` INT NOT NULL,
    `old_status` ENUM('CREATED','CHECKED_IN','LOADED','IN_TRANSIT','CLAIMED','LOST'),
    `new_status` ENUM('CREATED','CHECKED_IN','LOADED','IN_TRANSIT','CLAIMED','LOST'),
    `changed_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT `fk_bh_baggage`
        FOREIGN KEY (`baggage_id`) REFERENCES `Baggage`(`baggage_id`)
        ON UPDATE CASCADE ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE INDEX `idx_bh_baggage` ON `Baggage_History` (`baggage_id`);
